📊 CV Downloads Tracking Table Setup

📋 Instructions:

  1. Go to your Supabase Dashboard
  2. Navigate to SQL Editor
  3. Copy the SQL code below
  4. Paste and execute it
  5. This will create the cv_downloads table for tracking downloads

🗄️ SQL Code to Execute:

-- Create cv_downloads table for tracking CV downloads CREATE TABLE IF NOT EXISTS public.cv_downloads ( id SERIAL PRIMARY KEY, shopkeeper_id UUID NOT NULL, cv_id INTEGER NOT NULL, download_type VARCHAR(20) NOT NULL DEFAULT 'pdf', downloaded_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(), file_size_kb INTEGER, ip_address INET, user_agent TEXT, created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW() ); -- Add foreign key constraints ALTER TABLE public.cv_downloads ADD CONSTRAINT fk_cv_downloads_shopkeeper FOREIGN KEY (shopkeeper_id) REFERENCES auth.users(id) ON DELETE CASCADE; -- Create indexes for better performance CREATE INDEX IF NOT EXISTS idx_cv_downloads_shopkeeper_id ON public.cv_downloads(shopkeeper_id); CREATE INDEX IF NOT EXISTS idx_cv_downloads_cv_id ON public.cv_downloads(cv_id); CREATE INDEX IF NOT EXISTS idx_cv_downloads_downloaded_at ON public.cv_downloads(downloaded_at); -- Enable Row Level Security (RLS) ALTER TABLE public.cv_downloads ENABLE ROW LEVEL SECURITY; -- Create RLS policies -- Shopkeepers can only see their own downloads CREATE POLICY "Shopkeepers can view own downloads" ON public.cv_downloads FOR SELECT USING (shopkeeper_id = auth.uid()); -- Shopkeepers can insert their own downloads CREATE POLICY "Shopkeepers can insert own downloads" ON public.cv_downloads FOR INSERT WITH CHECK (shopkeeper_id = auth.uid()); -- Admins can see all downloads CREATE POLICY "Admins can view all downloads" ON public.cv_downloads FOR SELECT USING ( EXISTS ( SELECT 1 FROM auth.users WHERE auth.users.id = auth.uid() AND auth.users.raw_user_meta_data->>'role' = 'admin' ) ); -- Create a view for download statistics CREATE OR REPLACE VIEW public.download_stats AS SELECT shopkeeper_id, COUNT(*) as total_downloads, COUNT(DISTINCT cv_id) as unique_cvs_downloaded, MIN(downloaded_at) as first_download, MAX(downloaded_at) as last_download, SUM(file_size_kb) as total_size_kb FROM public.cv_downloads GROUP BY shopkeeper_id; -- Grant permissions GRANT SELECT ON public.download_stats TO authenticated; GRANT SELECT, INSERT ON public.cv_downloads TO authenticated;

💰 Billing Features:

  • Download Tracking: Every CV download is recorded with timestamp
  • Shopkeeper Identification: Each download is linked to the shopkeeper
  • File Size Tracking: Track data usage for billing
  • Download Statistics: View total downloads per shopkeeper
  • Date Range Queries: Filter downloads by date for billing periods

📈 Usage Examples:

Get total downloads for a shopkeeper:

SELECT total_downloads, unique_cvs_downloaded, first_download, last_download FROM public.download_stats WHERE shopkeeper_id = 'your-shopkeeper-uuid';

Get downloads for billing period:

SELECT COUNT(*) as downloads_this_month FROM public.cv_downloads WHERE shopkeeper_id = 'your-shopkeeper-uuid' AND downloaded_at >= '2024-01-01' AND downloaded_at < '2024-02-01';
